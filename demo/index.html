How it works

  1. Load rankings from localStorage — best endpoints tried first
  2. Golden shortcut — if top source score ≥ 30, try it alone (1 request)
  3. Race top-8 endpoints in parallel (fast tier, 2.5 s timeout)
  4. Fallback to the full 85+ pool if fast tier fails (6 s timeout)
  5. Winner scores +2; all scores decay × 0.92 each session
  6. Cache result in memory for 30 s — subsequent calls are instant

// <script src="https://cdn.jsdelivr.net/gh/DosX-dev/FastIP.js@latest/src/FastIP.js"></script>

// Promise-style
FastIP.getPublicIP()
  .then(function(ip) {
    console.log("Your IP:", ip);
  })
  .catch(function() {
    console.log("Could not detect IP");
  });

// async / await
const ip = await FastIP.getPublicIP();
console.log("Your IP:", ip);
// Returns { v4: Promise<string|null>, v6: Promise<string|null> }
const { v4, v6 } = FastIP.getDualStack();

v4.then(function(ip) {
  console.log("IPv4:", ip);
});

v6.then(function(ip) {
  // null if client has no IPv6 connectivity
  console.log("IPv6:", ip ?? "not available");
});
